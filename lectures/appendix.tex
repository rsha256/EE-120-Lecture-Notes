\subsection{Properties of the DTFT}

\[
    x(n)=\frac{1}{2 \pi} \int_{\langle 2 \pi\rangle} X(\omega) e^{i \omega n} d \omega \longleftrightarrow X(\omega)=\sum_{n=-\infty}^{\infty} x(n) e^{-i \omega n}
\]
